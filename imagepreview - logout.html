import random import micropip import asyncio from pyodide.ffi import create_proxy from js import FileReader, URL async def main(): await micropip.install("pillow") from PIL import Image, ImageOps # Hardcoded username and password VALID_USERNAME = "user" VALID_PASSWORD = "password" # Function to validate login def handle_login(event): username_input = Element("username").element.value password_input = Element("password").element.value login_output = Element("login-output") analysis_section = Element("analysis-section") if username_input == VALID_USERNAME and password_input == VALID_PASSWORD: login_output.write("Login successful!") Element("login-section").element.style.display = "none" analysis_section.element.style.display = "block" else: login_output.write("Invalid username or password. Please try again.") # Blood pressure generation function def generate_blood_pressure(): systolic = random.randint(90, 180) # Simulate systolic value diastolic = random.randint(60, 100) # Simulate diastolic value return f"{systolic}/{diastolic} mmHg" def display_blood_pressure(event): bp_output = Element("bp-output") blood_pressure = generate_blood_pressure() bp_output.write(f"Generated Blood Pressure: {blood_pressure}") # Image analysis function def analyze_image(file_input): analysis_output = Element("analysis-output") if file_input: filename = file_input.name.lower() # Check for cancer-related keywords in the filename if "cancer" in filename: analysis_output.element.innerHTML = 'Result: The image name suggests it indicates skin cancer.' elif "mild" in filename: analysis_output.element.innerHTML = 'Result: The image name suggests it indicates mild cancer.' else: analysis_output.element.innerHTML = 'Result: The image name suggests it does not indicate cancer.' else: analysis_output.write("No image uploaded.") # Attach event listeners Element("login-btn").element.onclick = handle_login Element("generate-btn").element.onclick = display_blood_pressure # Attach event listener for image input def handle_image_upload(event): files = event.target.files # Get the files from the event if len(files) > 0: file_input = files.item(0) # Get the first file from the file input analyze_image(file_input) # Pass the file input to analyze_image # Generate an Object URL for the selected image reader = FileReader.new() reader.onloadend = create_proxy(lambda e: update_image_preview(reader.result)) reader.readAsDataURL(file_input) else: Element("analysis-output").write("Failed to upload file.") # Error message def update_image_preview(image_url): img_preview = Element("img-preview") img_preview.element.src = image_url # Set image preview source Element("upload-img").element.onchange = handle_image_upload # Logout functionality def handle_logout(event): Element("login-section").element.style.display = "block" Element("analysis-section").element.style.display = "none" Element("login-output").write("Logged out successfully.") Element("username").element.value = "" Element("password").element.value = "" Element("logout-btn").element.onclick = handle_logout # Run the main coroutine asyncio.ensure_future(main())

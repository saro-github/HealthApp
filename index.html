import random import micropip import asyncio from js import window # Import 'window' to use JavaScript's URL.createObjectURL async def main(): await micropip.install("pillow") from PIL import Image, ImageOps # Hardcoded username and password VALID_USERNAME = "user" VALID_PASSWORD = "password" # Function to validate login def handle_login(event): username_input = Element("username").element.value password_input = Element("password").element.value login_output = Element("login-output") analysis_section = Element("analysis-section") logout_btn = Element("logout-btn") if username_input == VALID_USERNAME and password_input == VALID_PASSWORD: login_output.write("Login successful!") Element("login-section").element.style.display = "none" analysis_section.element.style.display = "block" logout_btn.element.style.display = "block" # Show the logout button else: login_output.write("Invalid username or password. Please try again.") # Blood pressure generation function def generate_blood_pressure(): systolic = random.randint(90, 140) diastolic = random.randint(60, 90) return f"{systolic}/{diastolic} mmHg" def display_blood_pressure(event): bp_output = Element("bp-output") blood_pressure = generate_blood_pressure() bp_output.write(f"Generated Blood Pressure: {blood_pressure}") # Image analysis function def analyze_image(file_input): analysis_output = Element("analysis-output") preview_image = Element("img-preview") # Display the image preview using JavaScript's URL.createObjectURL preview_image.element.src = window.URL.createObjectURL(file_input) if file_input: filename = file_input.name # Check filename for "cancer" if "cancer" in filename.lower(): analysis_output.element.innerHTML = 'Result: The image name suggests it indicates skin cancer.' elif "mild" in filename.lower(): analysis_output.element.innerHTML = 'Result: The image name suggests it indicates mild skin cancer.' else: analysis_output.element.innerHTML = 'Result: The image name suggests no skin cancer.' else: analysis_output.write("No image uploaded.") # Attach event listeners Element("login-btn").element.onclick = handle_login Element("generate-btn").element.onclick = display_blood_pressure # Attach event listener for image input def handle_image_upload(event): files = event.target.files if len(files) > 0: file_input = files.item(0) analyze_image(file_input) else: Element("analysis-output").write("Failed to upload file.") Element("upload-img").element.onchange = handle_image_upload # Run the main coroutine asyncio.ensure_future(main())
